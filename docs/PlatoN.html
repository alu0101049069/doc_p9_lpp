    # File lib/comida/plato.rb, line 90
def porc_glucidos
  resultado = 0
  peso = 0
  posicion = @listali.cabeza
  

  while(posicion != nil)
        resultado = (resultado + posicion.value.glucidos)
        posicion = posicion.sig
  end
  posicion = @listagr.cabeza

  while(posicion != nil)
        peso = (peso + posicion.value)
        posicion = posicion.sig
  end
  
  
  
  return (resultado/peso)*100

end

def to_se
  resultados =   @nombre 
  posicion = @listali.cabeza
 
  while(posicion != nil)

  end

 return resultados
